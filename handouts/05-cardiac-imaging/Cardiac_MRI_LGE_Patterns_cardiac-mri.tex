\begin{importantbox}
\textbf{LGE 判讀核心原則}
\begin{itemize}[leftmargin=*]
    \item \textbf{缺血性 (Ischemic)}：從心內膜開始，沿冠狀動脈支配區域分布
    \item \textbf{非缺血性 (Non-ischemic)}：不沿冠狀動脈分布，可為中層、心外膜或瀰漫性
\end{itemize}
\end{importantbox}

\section{Late Gadolinium Enhancement 基礎原理}
%============================================================

\subsection{什麼是 LGE？}

Late Gadolinium Enhancement（延遲顯影強化）是心臟磁振造影中最重要的組織特徵化技術之一。

\begin{longtable}{p{4cm}p{10cm}}
\toprule
\textbf{項目} & \textbf{說明} \\
\midrule
\endfirsthead
\bottomrule
\endfoot
原理\newline (Mechanism) & Gadolinium 對比劑會積聚在細胞外間質擴大的區域，如纖維化、壞死或水腫組織 \\
\midrule
掃描時機\newline (Timing) & 注射對比劑後 10-15 分鐘進行掃描，此時正常心肌已排除對比劑，而病變區域仍有滯留 \\
\midrule
影像表現\newline (Appearance) & 病變區域呈現\textbf{高信號（亮白色）}，正常心肌為\textbf{低信號（暗黑色）} \\
\midrule
臨床意義\newline (Clinical significance) & 偵測心肌纖維化、梗塞、發炎，評估心肌存活性 (viability)，預測預後 \\
\bottomrule
\end{longtable}

\subsection{LGE 判讀流程}

\begin{enumerate}[leftmargin=*]
    \item 確認影像品質：nulling 是否適當，正常心肌應為暗黑色
    \item 辨識強化區域：高信號區域的位置與分布
    \item 判斷強化模式：內膜下、透壁、中層、心外膜、瀰漫性
    \item 對照冠狀動脈分布：是否符合特定冠狀動脈支配區域
    \item 結合臨床病史：缺血性 vs 非缺血性心肌病變
\end{enumerate}

\section{缺血性心肌病變 (Ischemic Patterns)}
%============================================================

\note{
\textbf{關鍵特徵}：LGE 從心內膜 (endocardium) 開始，沿冠狀動脈支配區域 (coronary territory) 分布，符合「wavefront phenomenon」（波前現象）
}

\subsection{內膜下梗塞 (Sub-Endocardial Infarct)}

\begin{longtable}{p{4cm}p{10cm}}
\toprule
\textbf{特徵} & \textbf{說明} \\
\midrule
\endfirsthead
\bottomrule
\endfoot
LGE 分布 & 僅侷限於心內膜下區域，未達透壁 \\
\midrule
透壁程度\newline (Transmurality) & $<$50\% 壁厚 \\
\midrule
臨床意義 & 較早期或較小範圍的梗塞\newline 該區域心肌仍有\textbf{存活性 (viability)}，血管重建可能有益 \\
\midrule
鑑別重點 & 必須符合單一冠狀動脈支配區域 \\
\bottomrule
\end{longtable}

\subsection{透壁性梗塞 (Transmural Infarct)}

\begin{longtable}{p{4cm}p{10cm}}
\toprule
\textbf{特徵} & \textbf{說明} \\
\midrule
\endfirsthead
\bottomrule
\endfoot
LGE 分布 & 從心內膜延伸至心外膜，完全穿透心室壁 \\
\midrule
透壁程度\newline (Transmurality) & $>$50\% 壁厚，通常接近 100\% \\
\midrule
臨床意義 & 較嚴重或較晚期的梗塞\newline 該區域心肌\textbf{無存活性}，血管重建無法改善功能 \\
\midrule
預後意義 & LGE 範圍越大，心室功能恢復機會越低，預後越差 \\
\bottomrule
\end{longtable}

\keyfinding{
\textbf{存活性評估 (Viability Assessment)}\\
\begin{itemize}[leftmargin=*]
    \item LGE $<$25\% 壁厚：血管重建後功能恢復機會高
    \item LGE 25-50\% 壁厚：可能部分恢復
    \item LGE $>$50\% 壁厚：功能恢復機會低
\end{itemize}
}

%============================================================
\section{非缺血性心肌病變 (Non-Ischemic Patterns)}
%============================================================

\note{
\textbf{關鍵特徵}：LGE \textbf{不}沿冠狀動脈支配區域分布，可能出現在中層、心外膜、RV 接合點，或呈瀰漫性分布
}

\subsection{中層強化 (Mid-Wall Enhancement)}

\begin{longtable}{p{4cm}p{10cm}}
\toprule
\textbf{相關疾病} & \textbf{特徵說明} \\
\midrule
\endfirsthead
\bottomrule
\endfoot
擴張型心肌病變\newline (Dilated Cardiomyopathy, DCM) & 中隔中層線性強化，常見於 interventricular septum\newline 預後不良指標，與心律不整風險相關 \\
\midrule
心肌炎\newline (Myocarditis) & 中層或心外膜強化\newline 常見於 inferolateral wall\newline 急性期可合併心肌水腫 (T2 高信號) \\
\midrule
類肉瘤\newline (Sarcoidosis) & 多發性、斑塊狀中層或心外膜強化\newline 常伴隨局部室壁運動異常\newline 可能累及右心室 \\
\midrule
Anderson-Fabry 病 & 典型位於 basal inferolateral wall 中層\newline 常合併左心室肥厚 \\
\midrule
Chagas 病 & 心尖部及下壁中層強化\newline 常見於中南美洲流行區 \\
\bottomrule
\end{longtable}

\subsection{心外膜強化 (Epicardial Enhancement)}

\begin{longtable}{p{4cm}p{10cm}}
\toprule
\textbf{相關疾病} & \textbf{特徵說明} \\
\midrule
\endfirsthead
\bottomrule
\endfoot
心肌炎\newline (Myocarditis) & 心外膜或中層強化\newline 典型位於 inferolateral wall\newline 病毒性心肌炎最常見 \\
\midrule
類肉瘤\newline (Sarcoidosis) & 可呈心外膜或中層分布\newline 常為多發性 \\
\midrule
Anderson-Fabry 病 & 可延伸至心外膜 \\
\midrule

\midrule
\endfirsthead
\bottomrule
\endfoot
肥厚型心肌病變\newline (HCM) & 前後 RV insertion points 強化\newline 與纖維化程度相關\newline 可能與心律不整風險有關 \\
\midrule

\subsection{瀰漫性/全域性強化 (Diffuse/Global Enhancement)}

\begin{longtable}{p{4cm}p{10cm}}
\toprule
\textbf{相關疾病} & \textbf{特徵說明} \\
\midrule
\endfirsthead
\bottomrule
\endfoot
心臟類澱粉沉積症\newline (Cardiac Amyloidosis) & 瀰漫性心內膜下或透壁強化\newline 傳統 LGE 技術可能難以偵測（因正常心肌也受影響）\newline \textbf{T1 mapping} 更為敏感：native T1 顯著升高 \\

\end{longtable}

\section{重點整理與臨床建議}
%============================================================

\begin{importantbox}
\textbf{LGE 判讀重點}
\begin{enumerate}[leftmargin=*]
    \item \textbf{先判斷缺血 vs 非缺血}：內膜下起始 + 冠狀動脈分布 = 缺血性
    \item \textbf{評估透壁程度}：影響存活性評估與治療決策
    \item \textbf{結合臨床病史}：胸痛史、心肌酵素、冠狀動脈造影結果
    \item \textbf{注意特殊模式}：中層強化考慮 DCM/myocarditis，RV insertion 考慮 HCM
    \item \textbf{搭配其他序列}：T1/T2 mapping, STIR 等進一步鑑別
\end{enumerate}
\end{importantbox}

\subsection{何時懷疑非缺血性病因？}

\begin{itemize}[leftmargin=*]
    \item LGE 不沿冠狀動脈分布
    \item 年輕患者無明顯心血管危險因子
    \item 強化位於中層或心外膜
    \item 多發性斑塊狀強化
    \item 冠狀動脈造影正常
\end{itemize}
